6. DAG BUILDER

This resolves execution order.

from collections import defaultdict, deque
def build_dag(tasks):
    graph = defaultdict(list)
    indeg = defaultdict(int)
    for t in tasks.values():
        for d in t.deps:
            graph[d].append(t.name)
            indeg[t.name] += 1
    queue = deque([t for t in tasks if indeg[t] == 0])
    order = []
    while queue:
        node = queue.popleft()
        order.append(node)
        for nxt in graph[node]:
            indeg[nxt] -= 1
            if indeg[nxt] == 0:
                queue.append(nxt)
    return order

This gives you a topological execution order.

from concurrent.futures import ThreadPoolExecutor
def execute(tasks, order):
    state = load_state()
    def run_task(name):
        task = tasks[name]
        if state.get(name, {}).get("done"):
            print(f"[SKIP] {name}")
            return
        print(f"[RUN] {name}")
        task.run()
        state[name] = {"done": True}
        save_state(state)
    with ThreadPoolExecutor(max_workers=4) as pool:
        for name in order:
            pool.submit(run_task, name)
